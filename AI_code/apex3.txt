public class LinearSearch {
    
    public static void searchInteger(List<Integer> myList, Integer target) {
        Boolean found = false;
        
        for(Integer num : myList) {
            if(num == target) {
                found = true;
                break;
            }
        }
        
        if(found) {
            System.debug('Element found.');
        } else {
            System.debug('Element not found.');
        }
    }
}
